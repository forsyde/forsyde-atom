\newsavebox\insig

  \begin{ct-signal}[xmin=0, xmax=10, ymin=-1.5, ymax=1.5]{2}

\newsavebox\outsig
\begin{lrbox}{\outsig}
  \begin{ct-signal}[xmin=0, xmax=10, ymin=-1.5, ymax=1.5]{2}

    \addplot[domain=2:5,mark=none,samples=50,blue] {-sin(deg(x))};

    \addplot[domain=8:11,mark=none,samples=50,blue] {-sin(deg(x))};
    \addplot[mark=none,blue] coordinates {(2,{sin(deg(2))}) (2,{-sin(deg(2))})};
    \addplot[mark=none,blue] coordinates {(5,{sin(deg(5))}) (5,{-sin(deg(5))})};
    \addplot[mark=none,blue] coordinates {(8,{sin(deg(8))}) (8,{-sin(deg(8))})};

    \end{ct-signal} 
\end{lrbox}

  \signal[] (c1-p.o1) -> (o1l);
