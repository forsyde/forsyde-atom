\documentclass{standalone}
\usepackage{forsyde-tikz}
\usepackage{forsyde-plot}

\def\sigsin(#1){sin(10 * deg(#1))}
\def\dsigsin(#1){sin(10 * deg(#1) - deg(2))}
\begin{document}

\newsavebox\insig
\begin{lrbox}{\insig}
  \begin{ct-signal}[xmin=0, xmax=1, ymin=-1.5, ymax=1.5]{2}
    \ctsignal{$S_{i,1}$}\ctf{0}{1}{\sigsin}\ctend
  \end{ct-signal} 
\end{lrbox}


\newsavebox\outsig
\begin{lrbox}{\outsig}
  \begin{ct-signal}[xmin=0, xmax=1, ymin=-1.5, ymax=1.5]{2}
    \ctsignal{$S_{o,1}$}\ctct{0}{.2}{0}\ctf{.2}{1}{\dsigsin}\ctend
    \end{ct-signal} 
\end{lrbox}


\begin{tikzpicture}[]
  \leafstd[moc=ct, ni=1, no=1, nf=2, f1={$0.2$}, f2={$\lambda x \rightarrow 0$}, type=delay]{c1} {0,0} {}; 
  \node[anchor=east, scale=.7] (i1) at ($(c1)-(3,0)$) {\usebox{\insig}}; 
  \node[anchor=west, scale=.7] (o1) at ($(c1)+(3,0)$) {\usebox{\outsig}};
  \node[anchor=east] (i1l) at ($(c1-p.i1)-(1,0)$) {$S_{i,1}$};
  \node[anchor=west] (o1l) at ($(c1-p.o1)+(1,0)$) {$S_{o,1}$};

  \signal[] (i1l) -> (c1-p.i1); 
  \signal[] (c1-p.o1) -> (o1l);
\end{tikzpicture}  
\end{document}
