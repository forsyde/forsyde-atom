\newsavebox\network
\begin{lrbox}{\network}
\begin{tikzpicture}[nomoccolor, type style = \scriptsize\texttt, global scale = .5]
  \vector[] (0,.5) - (0,4); \vector[] (2,.5) - (2,4);
  \foreach \i in {1,...,4} {
    \pgfmathsetmacro{\j}{\i - .5}
    \signal[token=scalar] (-1,\i) -> (0,\i);
    \signal[token=scalar] (-1,\j) -> (0,\j);
    \signal[] (0,\i) -> (2,\i);
    \signal[] (0,\j) -> (2,\i);
    \signal[token=scalar-scalar] (2,\i) -> (3,\i);
  }
\end{tikzpicture}
\end{lrbox}
